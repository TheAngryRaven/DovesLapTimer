16// On classic AVR (Mega, Uno) `double` is a 32-bit float (~7 significant
17// digits). At real-world latitudes/longitudes that quantizes position to
18// roughly 0.2-0.75 m and pushes per-fix haversine half-angle differences
19// below float precision: lap *counting* still works (the 7 m crossing
20// threshold absorbs it) but odometer increments and crossing interpolation
21// carry large relative error. The full advertised precision requires a
22// target with true 64-bit double (nRF52840, ESP32, SAMD51, RP2040, ...).
23// See README "Hardware" notes.
24#if defined(__SIZEOF_DOUBLE__) && (__SIZEOF_DOUBLE__ < 8)
25#warning "DovesLapTimer: 'double' is only 32 bits on this target (classic AVR). Lap counting works but GPS math runs degraded - distances and interpolated crossing times will be noticeably less accurate. Use a 64-bit-double MCU (e.g. XIAO nRF52840) for full precision."
26#endif

70// GPS time base: milliseconds since UTC midnight, wraps 86,399,999 -> 0
71#define DOVES_MILLIS_PER_DAY 86400000UL

73// GPS input validation: a single-fix jump beyond this is treated as a glitch
74// and dropped; after GPS_JUMP_REACCEPT_COUNT consecutive far fixes the new
75// position is accepted as real (signal re-acquisition) and the position is
76// re-seeded without crediting the gap to the odometer.
77#define GPS_MAX_PLAUSIBLE_JUMP_METERS 500.0
78#define GPS_JUMP_REACCEPT_COUNT 3
79
80// Max believable gap between the two consecutive buffered fixes that straddle
81// a crossing line. A larger gap means the GPS time base stepped (e.g. u-blox
82// re-acquisition) and the pair cannot be interpolated coherently.
83#define CROSSING_MAX_FIX_GAP_MS 10000UL
84
85// The straddling pair's summed distance to the line is validated against
86// max(crossingThresholdMeters, factor * pair spacing) so that low-rate GPS
87// (widely spaced fixes) doesn't fail validation purely on sample density.
88#define CROSSING_PAIR_SPACING_FACTOR 1.25
